void LauncherCore::launchGame(const GameSettings& settings,
                      const Account& account) {
    // Resolve metadata before starting the Minecraft process
    auto command = buildLaunchCommand(settings, account);
    prepareNatives(settings.versionId);
    startProcess(settings.javaPath, command);
}
JavaVersion JavaManager::findSuitableJava(const QString& mcVersion,
                                              bool allowDownload) {
    const int required = requiredJavaMajor(mcVersion);
    auto installed = scanInstalledJava();
    return bestMatchingRuntime(installed, required, allowDownload);
}
QString AccountsTab::generateOfflineUuid(const QString& username) const {
    // Vanilla-compatible OfflinePlayer UUID (MD5 v3 + RFC variant bits)
    return minecraftOfflineUuid(
        QString("OfflinePlayer:%1").arg(username));
}

Quick fixes

Game crashes / out of memory

Raise -Xmx in Settings — about 4-6 GB for modpacks, 2-4 GB for vanilla. Never assign more than ~75% of your RAM.

Modded version won't launch

Its vanilla base must be installed. Reinstall the build, or open the Console tab — it prints the exact reason.

Low FPS in game

Install Sodium (Fabric) or OptiFine (Forge), add Iris for shaders, and give the instance a bit more memory.
